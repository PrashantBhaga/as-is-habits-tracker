    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Google Sheets Habit Tracker</title>

    <style>

        /* ... (previous styles remain the same) ... */

        #loginSection, #habitSection {

            display: none;

        }

    </style>

</head>

<body>

    <div class="container">

        <h2>Google Sheets Habit Tracker</h2>

        

        <div id="loginSection">

            <input type="text" id="userIdInput" class="input-field" placeholder="Enter your user ID">

            <button onclick="login()">Login</button>

        </div>


        <div id="habitSection">

            <input type="text" id="habitInput" class="input-field" placeholder="What was the habit?">

            <div class="helper-text">Does this behavior help me become the person I wish to be?</div>

            <div class="btn-container">

                <button class="btn btn-positive" onclick="setHabitType('+')">+</button>

                <button class="btn btn-neutral" onclick="setHabitType('=')">=</button>

                <button class="btn btn-negative" onclick="setHabitType('-')">-</button>

            </div>

            <input type="text" id="cueInput" class="input-field" placeholder="What was the cue?">

            <button id="submitBtn" onclick="submitHabit()">Submit</button>

            <button onclick="viewHabits()">View My Habits</button>

            <button onclick="logout()">Logout</button>

        </div>


        <div id="message"></div>

        <div id="habitList"></div>

    </div>


    <script>

        var habitType = null;

        var userId = null;

        var GOOGLE_SCRIPT_URL = https://script.google.com/macros/s/AKfycbwYxKdZcnpxzDc4ginEQPkHKiCh6mQE-USEjedGoQjJI45fb0bZvumdPZkF91VR_Q7f/exec;


        function login() {

            userId = document.getElementById('userIdInput').value;

            if (userId) {

                document.getElementById('loginSection').style.display = 'none';

                document.getElementById('habitSection').style.display = 'block';

            } else {

                alert('Please enter a user ID');

            }

        }


        function logout() {

            userId = null;

            document.getElementById('loginSection').style.display = 'block';

            document.getElementById('habitSection').style.display = 'none';

            document.getElementById('userIdInput').value = '';

        }


        function setHabitType(type) {

            habitType = type;

            var buttons = document.querySelectorAll('.btn');

            for (var i = 0; i < buttons.length; i++) {

                buttons[i].style.opacity = '0.5';

            }

            event.target.style.opacity = '1';

        }


        function submitHabit() {

            if (!habitType) {

                alert("Please select a habit type (+, =, or -)");

                return;

            }


            var habit = document.getElementById('habitInput').value;

            var cue = document.getElementById('cueInput').value;


            var data = {

                habit: habit,

                type: habitType,

                cue: cue,

                userId: userId

            };


            fetch(GOOGLE_SCRIPT_URL, {

                method: 'POST',

                body: JSON.stringify(data)

            })

            .then(response => response.json())

            .then(result => {

                if (result.success) {

                    document.getElementById('message').innerText = "Habit logged successfully!";

                } else {

                    document.getElementById('message').innerText = "Error logging habit.";

                }

            })

            .catch(error => {

                console.error('Error:', error);

                document.getElementById('message').innerText = "Error logging habit.";

            });


            // Reset form

            habitType = null;

            document.getElementById('habitInput').value = '';

            document.getElementById('cueInput').value = '';

            var buttons = document.querySelectorAll('.btn');

            for (var i = 0; i < buttons.length; i++) {

                buttons[i].style.opacity = '1';

            }

        }


        function viewHabits() {

            fetch(GOOGLE_SCRIPT_URL + '?userId=' + userId)

            .then(response => response.json())

            .then(data => {

                var habitList = document.getElementById('habitList');

                habitList.innerHTML = '<h3>Your Habits:</h3>';

                data.forEach(habit => {

                    habitList.innerHTML += `<p>${habit[0]}: ${habit[1]} (${habit[2]})</p>`;

                });

            })

            .catch(error => {

                console.error('Error:', error);

                document.getElementById('message').innerText = "Error retrieving habits.";

            });

        }

    </script>

</body>
